// Système de timeline pour les expériences professionnelles function showExperience(companyId, index) { // Masquer toutes les cartes document.querySelectorAll('.experience-detail-card').forEach(card => { card.classList.remove('active'); card.classList.add('hidden'); }); // Afficher la carte sélectionnée const targetCard = document.getElementById('exp-' + companyId); if (targetCard) { targetCard.classList.remove('hidden'); // Petit délai pour permettre la transition setTimeout(() => { targetCard.classList.add('active'); }, 10); } // Mettre à jour les points de timeline document.querySelectorAll('.timeline-point').forEach((point, idx) => { point.classList.toggle('active', idx === index); const dot = point.querySelector('.timeline-dot-inner'); const label = point.querySelector('span'); if (idx === index) { dot.style.background = 'var(--cyan)'; dot.style.borderColor = 'var(--cyan)'; label.style.color = 'var(--cyan)'; label.style.fontWeight = '600'; } else { if (idx === 6) { // Integral Tech - dernier point gris dot.style.background = 'var(--dark)'; dot.style.borderColor = '#4B5563'; label.style.color = '#6B7280'; } else { dot.style.background = 'var(--dark)'; dot.style.borderColor = 'var(--cyan)'; label.style.color = '#9CA3AF'; } label.style.fontWeight = '500'; } }); } // Initialiser avec HT241 actif au chargement document.addEventListener('DOMContentLoaded', () => { showExperience('ht241', 0); });
